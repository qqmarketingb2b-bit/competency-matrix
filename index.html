Шаг 3. Вставь код ниже в редактор (замени всё что там есть):
const SHEET_NAME = 'Ответы';

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName(SHEET_NAME);
    if (!sheet) {
      sheet = ss.insertSheet(SHEET_NAME);
      sheet.appendRow(['id','name','email','role','cluster','months',
        'submittedAt','status','matrix',
        'soft_0','soft_1','soft_2','soft_3','soft_4','soft_5',
        'cluster_0','cluster_1','cluster_2','cluster_3','cluster_4',
        'cluster_5','cluster_6','cluster_7','cluster_8',
        'open_1','open_2','open_3']);
    }
    if (data.action === 'submit') {
      const sa = data.softAnswers || {};
      const ca = data.clusterAnswers || {};
      const oa = data.openAnswers || {};
      sheet.appendRow([
        data.id, data.name, data.email,
        data.role?.name, data.role?.cluster, data.months,
        data.submittedAt, 'new', '',
        sa[0]||'',sa[1]||'',sa[2]||'',sa[3]||'',sa[4]||'',sa[5]||'',
        ca[0]||'',ca[1]||'',ca[2]||'',ca[3]||'',ca[4]||'',
        ca[5]||'',ca[6]||'',ca[7]||'',ca[8]||'',
        oa[1]||'',oa[2]||'',oa[3]||''
      ]);
      return ok({saved: true});
    }
    if (data.action === 'getAll') {
      const rows = sheet.getDataRange().getValues();
      const headers = rows[0];
      const records = rows.slice(1).map(r => {
        const obj = {};
        headers.forEach((h,i) => obj[h] = r[i]);
        return obj;
      });
      return ok(records);
    }
    if (data.action === 'update') {
      const rows = sheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        if (String(rows[i][0]) === String(data.id)) {
          if (data.matrix !== undefined)
            sheet.getRange(i+1, 9).setValue(data.matrix);
          if (data.status !== undefined)
            sheet.getRange(i+1, 8).setValue(data.status);
          break;
        }
      }
      return ok({updated: true});
    }
    return ok({error: 'unknown action'});
  } catch(err) {
    return ok({error: err.message});
  }
}

function doGet(e) {
  return ok({status: 'ok'});
}

function ok(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
